## Usage

<DocBlockContainer>
  <Heading style={{ color: tokens.colors.blue[900] }}>My app card</Heading>
  <AppCard accentColor={tokens.colors.yellow[700]}>
    <Centered>
      <Spinner spinnerColor={tokens.colors.red[600]} />
    </Centered>
  </AppCard>
</DocBlockContainer>

```tsx
import { AppCard, Centered, Heading, Spinner, tokens } from 'mirotone-react';

function App() {
  return (
    <>
      <Heading style={{ color: tokens.colors.blue[900] }}>My app card</Heading>
      <AppCard accentColor={tokens.colors.yellow[700]}>
        <Centered>
          <Spinner spinnerColor={tokens.colors.red[600]} />
        </Centered>
      </AppCard>
    </>
  );
}
```
